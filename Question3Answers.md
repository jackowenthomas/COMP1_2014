FUNCTION GetPlayerName()
	ValidName == False
	PlayerName == INPUT("Please enter your name: ")
	WHILE LENGTH(PlayerName) = 0:
		PlayerName == INPUT("Please enter your name: ")
		ValidName == False
	ValidName == True
	RETURN PlayerName
